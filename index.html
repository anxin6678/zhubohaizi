/** * ========================================== * 纯网页接口音频引擎 (Web Audio API & 外链) * 完全不需要本地上传音效文件！ * ========================================== */ const AudioSys = { // 1. BGM: 使用外部网络直链 (这里我放了一个免费的复古电子节拍做测试) // 你可以随时把它换成网上找到的任意 MP3 链接 bgm: new Audio('https://cdn.pixabay.com/download/audio/2022/01/18/audio_d0a13f69d2.mp3'), // 初始化网页音频合成器接口 audioCtx: null, initBGM: function() { this.bgm.loop = true; this.bgm.volume = 0.3; this.bgm.play().catch(e => console.warn("外链BGM可能被浏览器拦截或链接失效:", e)); // 激活音频上下文 (必须在玩家点击后激活) if (!this.audioCtx) { this.audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } if (this.audioCtx.state === 'suspended') { this.audioCtx.resume(); } }, // 2. 纯代码合成复古 8-bit 音效 playSFX: function(type) { if (!this.audioCtx) return; const osc = this.audioCtx.createOscillator(); const gainNode = this.audioCtx.createGain(); osc.connect(gainNode); gainNode.connect(this.audioCtx.destination); const now = this.audioCtx.currentTime; // 根据不同动作，用代码合成不同的频率和波形 switch(type) { case 'click': // 清脆的短促点击声 (正弦波) osc.type = 'sine'; osc.frequency.setValueAtTime(800, now); gainNode.gain.setValueAtTime(0.3, now); gainNode.gain.exponentialRampToValueAtTime(0.01, now + 0.1); osc.start(now); osc.stop(now + 0.1); break; case 'jine': // JINE 收发消息的“叮咚”声 (双重频率) osc.type = 'sine'; osc.frequency.setValueAtTime(1200, now); osc.frequency.setValueAtTime(1600, now + 0.1); gainNode.gain.setValueAtTime(0.4, now); gainNode.gain.linearRampToValueAtTime(0, now + 0.3); osc.start(now); osc.stop(now + 0.3); break; case 'stream': // 魔法变身开播特效声 (快速上升的琶音) osc.type = 'square'; osc.frequency.setValueAtTime(400, now); osc.frequency.linearRampToValueAtTime(800, now + 0.1); osc.frequency.linearRampToValueAtTime(1200, now + 0.2); osc.frequency.linearRampToValueAtTime(1600, now + 0.3); gainNode.gain.setValueAtTime(0.2, now); gainNode.gain.linearRampToValueAtTime(0, now + 0.4); osc.start(now); osc.stop(now + 0.4); break; case 'error': // 压力过载的报错声 (刺耳的锯齿波) osc.type = 'sawtooth'; osc.frequency.setValueAtTime(150, now); osc.frequency.linearRampToValueAtTime(100, now + 0.5); gainNode.gain.setValueAtTime(0.5, now); gainNode.gain.linearRampToValueAtTime(0.01, now + 0.5); osc.start(now); osc.stop(now + 0.5); break; } } };
